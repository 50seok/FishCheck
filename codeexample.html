@st.cache_resource(show_spinner="모델 로딩 중...")
def load_model() -> YOLO:
    if not Path(MODEL_PATH).exists():
        if not HF_REPO_ID:
            raise FileNotFoundError(...)
        from huggingface_hub import hf_hub_download
        Path(MODEL_PATH).parent.mkdir(
            parents=True, exist_ok=True
        )
        hf_hub_download(
            repo_id=HF_REPO_ID,
            filename="best.pt",
            local_dir="models"
        )
    return YOLO(MODEL_PATH)
★
@st.cache_resource — 반환값(YOLO 객체)을 서버 메모리에 고정. 페이지 새로고침·여러 사용자 접속 시에도 모델을 1회만 로딩
i
Path(...).exists() — 로컬 파일 먼저 확인 → 없을 때만 HF Hub 다운로드. 서버 재시작 시 중복 다운로드 방지
i
hf_hub_download — HF_REPO_ID/best.pt를 models/에 저장. Streamlit Cloud 환경에서 Git LFS 없이 모델 배포 가능
✓
return YOLO(MODEL_PATH) — .pt 가중치를 GPU/CPU에 올려 추론 준비 완료

def predict(img: Image.Image) -> dict:
    model   = load_model()
    results = model(img, verbose=False, conf=0.65)
    result  = results[0]

    if result.boxes is None or len(result.boxes) == 0:
        return {"detected": False, ...}

Line 71–79
seen  = set()
top3  = []
for b in boxes:
    name = result.names[b["cls"]]
    if name not in seen:
        seen.add(name)
        top3.append({
            "class_en" : name,
            "class_ko" : CLASS_KO.get(name, name),
            "confidence": b["conf"],
        })
    if len(top3) == 3:
        break
